// MyComponent.stories.js

import imageFile from './static/image.png';

export default {
  title: 'img'
};

const image = {
  src: imageFile,
  alt: 'my image'
};

export const withAnImage = () => ({
  template: `<img :src="src" :alt="alt"/>`,
  props: {
    src: {
      default: () => image.src,
    },
    alt: {
      default: () => image.alt,
    }
  }
});
